[ 迴圈 for ]
  • 用迴圈印出數字序列 33 , 36 , 39 , ... , 999 
    • for( int i=33 ; i <= 999 ; i=i+3 ){
	System.out.println( i ) ;
}
  • 用迴圈印出數字序列 88 , 77 , 66 , ... , 11 
    • for( int i=88 ; i >= 11 ; i=i-11 ){
	System.out.println( i ) ;
}
  • 用迴圈印出數字 57~67，遇到 63 用 continue 跳過 
    • for( int i=57 ; i <= 67 ; i=i+1 ){
	if( i==63 ){
		continue ;
	}
	System.out.println( i ) ;
}
  • 用迴圈找數字 380 ~ 400
    找到 17 的倍數，用 break 中斷搜尋 
    • for( int i=380 ; i <= 400 ; i=i+1 ){
	if( i%17==0 ){
		System.out.println( "found !! "+i ) ;
		break ;
	}		
}
[ 陣列、Array ]
  • 建立陣列，變數名稱 arr ，放入 3 , 7 , 2 , 4
    • int[] arr = new int[4] ;
arr[0] = 3 ;
arr[1] = 7 ;
arr[2] = 2 ;
arr[3] = 4 ;
  • 印出陣列內容如下：
    0 -> 3
    1 -> 7
    2 -> 2
    3 -> 4
    • for( int i=0 ; i < arr.length ; i++){
	System.out.println(i+" -> "+arr[i]);
}
  • 計算陣列總和，放入變數 total
    • int total = 0 ;
for( int i=0 ; i < arr.length ; i++){
	total = total + arr[i] ;
}
System.out.println( "total -> "+total );
  • 尋找陣列最大值，放入變數 max
    • int max = arr[0] ;
for( int i=0 ; i < arr.length ; i++){
	if( arr[i] > max ){
		max = arr[i] ;
	}
}
System.out.println( "max -> "+max );
[ 函式、Function ]
  • 建立函式 F1
    可以輸入 3 個數字，將 3 個數字相乘，丟出結果
    • // 建立
public static int F1( int a , int b ,int c){
	int ans = a*b*c ;
	return ans ;
}

// 使用
int r = F1( 3 ,4 ,5 );
  • 建立函式 F2
    可以輸入 2 個名字，將 2 個名字用逗號連接，作為結果丟出
    • // 建立
public static int F2( String name1 , String name2 ){
	String r = name1+","+name2 ;
	return r ;
}
// 使用
String result = F2( "Joe" , "May" ) ;

物件結構

  • 建立 class 「Student」
    # 具備變數：「name」放姓名、「math」放數學成績、「java」放 java 成績
    # 具備函式：show() 印出姓名、數學、java 成績 
    • class Student{
	
	String name ;
	int math ;
	int java ;

	public void show(){
		System.out.println("name : "+name ) ;
		System.out.println("math : "+math ) ;
		System.out.println("java : "+java ) ;
	}

}
  • 使用 Student class，建立 Student 物件
    姓名：Joe、數學成績：98、java 成績：99。並使用函式 show() 
    class Test{

	public static void main(String[] args){

		Student s = new Student() ;
		s.name = "Joe" ;
		s.math = 98 ;
		s.java = 99 ;
		s.show() ;
	}
}

應用語法片段

[ 文字 String ]
  • 用 charAt 搭配 for 取出文字變數 str 的每一個字元
    String str = "apple" ;
    印出內容如下：
    0 --> a
    1 --> p
    2 --> p
    3 --> l
    4 --> e 
    • String str = "apple" ;
for( int i=0 ; i < str.length() ; i++ ){
	char c = str.charAt(i) ;
	System.out.println( i+" --> "+ c ) ;
}
  • 用 charAt 搭配 for 取出文字變數 str 的每一個字元
    計算文字中出現字母 a 的次數 String str = "approach" ;
    • String str = "approach" ;
int total = 0 ;
for( int i=0 ; i < str.length() ; i++ ){
	char c = str.charAt(i) ;
	if( c == 'a' ){
		total++ ;
	}
}
System.out.println( total ) ;
  • 用 split 將 str 依據「,」把 str 切開，計算「晴天」次數
    String str = "晴天,雨天,晴天,晴天,雨天,晴天" ; 
    • String str = "晴天,雨天,晴天,晴天,雨天,晴天" ; 
String[] arr = str.split(",") ;
int total = 0 ;
for( int i=0 ; i < arr.length ; i++ ){
	if( arr[i].equals("晴天") == true ){
		total++ ;
	}
}
System.out.println( total ) ;

data[0][0] = "A" ; 
data[0][1] = "B" ; 
data[0][2] = "C" ; 
data[1][0] = "D" ; 
data[1][1] = "E" ; 
data[1][2] = "F" ;
