}

  • 數值比較組合 
    • int a = ? ;
int b = ? ;

// ----> && 且
if( a>5 && a<10 ){ 
    // 5 < a < 10
}
if( a%2==0 && a>20){
    // a 是大於 20 的偶數
}
if( a>0 && b>0 ){
    // a , b 都要 >0
}

// ----> || 或
if( a<5 || a>25 ){
    // a<5 或 a>25 
}
