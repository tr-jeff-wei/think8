基礎語法

  用 if / if-else

  • 把［積木］變成打字～



  把問題放 ( 框框 ) 裡

  • 框框裡只能放結果是 true 、false
  1. 可以直接放 true、false （通常沒有意義！） 
    2. // a. 機器人會掃地
if( true ){
	// 掃地
}

// b. 機器人不會掃地
if( false ){
	// 掃地
}
  2. 放結果是 true、false 的變數 
    3. var battery = 50 ;
var charge = ( battery < 30 ) ;
if( charge ){
	// 充電
}
  3. 放結果是 true、false 的條件式 
    4. // ex.1
var battery = 50 ;
if( battery < 30 ){
	// 充電
}

// ex.2：老闆是 Joe
var p = 'Jeff' ;
if( p=='Joe' ){
	// 老闆好
}else{
	// 歡迎光臨
}


  寫條件式（問問題） - 文字 

    var a = 'Jeff' ;
var b = 'Mary' ;
a == b  ; //  false
a != b  ; //  true

if( a==b ){
	// a , b 兩位名字一樣
}

if( a != 'Peter' ){
	// a 不是 Peter
}


  寫條件式（問問題） - 數字 

    var a = 5 ;
var b = 7 ;
var c = 9 ;

a == 5 ; // true
a != 5 ; // false
a >  b ; // false
a <  c ; // true
9 <= c ; // true

    // ex: 年紀 7 歲以上就可以上小學了
var age = 10 ;
if( age >= 7 ){
	// 可以上小學
}else{
	// 年紀太小了
}
